macro(common_add_unit_test name)
    set(USE_MOCK_COMMON FALSE)
    set(USE_MOCK_ENGINE TRUE)
    marabou_add_test(${COMMON_TESTS_DIR}/Test_${name} common USE_MOCK_COMMON USE_MOCK_ENGINE "unit")
endmacro()
